# Smart pane switching with awareness of Vim splits.
# See: https://github.com/christoomey/vim-tmux-navigator
is_vim="ps -o state= -o comm= -t '#{pane_tty}' | grep -iqE '^[^TXZ ]+ +(\\S+\\/)?g?(view|n?vim?x?)(diff)?$'"

# address vim mode switching delay (http://superuser.com/a/252717/65504)
set -s escape-time 0
# increase scrollback buffer size
set -g history-limit 50000
# tmux messages are displayed for 4 seconds
set -g display-time 4000
# refresh 'status-left' and 'status-right' more often
set -g status-interval 5

# focus events enabled for terminals that support them
set -g focus-events on
# super useful when using "grouped sessions" and multi-monitor setup
#setw -g aggressive-resize on

setw -g xterm-keys on  # Xterm Key Binding
set -g base-index 1  # Count Sessions from 1 on

# Title and Status Bar Settings
set -g set-titles off
#set -option -g set-titles-string ' [#S:#I #H] #W'
set -g status on
set -g status-position bottom  #set -g status-position [bottom | top]
set-option -g status-style bg=black  #set -g status-bg black
set -g status-fg white
set -g status-justify centre
set -g status-interval 5 
set -g status-left-length 50 #90
set -g status-right-length 50 #60
set -g status-left "#[fg=Green]#(whoami)#[fg=white]::#[fg=cyan]#{pane_current_path}#[fg=white]::#[fg=yellow]#(curl ipecho.net/plain;echo)#[fg=white]"
set -g status-right 'Batt: #{battery_icon} #{battery_percentage} #{battery_remain} | %a %h-%d %H:%M '
#set -g status-right '#[fg=Cyan]#S #[fg=white]%a %d %b %R'
setw -g window-status-current-format "#[fg=cyan,reverse]#{pane_current_path}::#W "
set-window-option -g automatic-rename off
#set -g bell-action any

# Tmux Continuum from Default Settings if needed
#set -g @continuum-save-interval '60'  #Default is 15 minutes
#set -g @continuum-save-interval '0'  #If you need Auto Saves Off a while

#==>Quick PANE RECYCLING

# ---KEY BINDINGS---
# easier and faster switching between next/prev window
bind C-p previous-window
bind C-n next-window
# CTRL-b then R sources dotfile with a message
bind-key R source-file ~/.tmux.conf \; display-message "Config reloaded..."
# Vim Key Bindings and which vim
bind-key -n C-h if-shell "$is_vim" "send-keys C-h"  "select-pane -L"
bind-key -n C-j if-shell "$is_vim" "send-keys C-j"  "select-pane -D"
bind-key -n C-k if-shell "$is_vim" "send-keys C-k"  "select-pane -U"
bind-key -n C-l if-shell "$is_vim" "send-keys C-l"  "select-pane -R"
bind-key -n C-\ if-shell "$is_vim" "send-keys C-\\" "select-pane -l"
set -g status-keys vi  # /opt/local/bin/vim
set -g mode-keys vi  # /opt/local/bin/vim

# Keybinding Optional SSH start with CTRL-b then S
bind-key S command-prompt "new-window -n %1 ’ssh %1’"

# PLUGINS -List of plugins Tmux Plugin Manager Install
set -g @plugin 'tmux-plugins/tpm'
set -g @plugin 'tmux-plugins/tmux-sensible'
set -g @plugin 'tmux-plugins/tmux-battery'
set -g @plugin 'tmux-plugins/tmux-resurrect'
set -g @plugin 'tmux-plugins/tmux-continuum'
set -g @plugin 'tmux-plugins/tmux-open'

# Initialize TMUX plugin manager (keep this line at the very bottom of tmux.conf)
run '~/.tmux/plugins/tpm/tpm'
